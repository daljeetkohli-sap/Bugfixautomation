  <script>
    const activeChips = { bugs: true, quality: true, report: true, security: true };

    function toggleChip(key) {
      activeChips[key] = !activeChips[key];
      document.getElementById('chip-' + key).classList.toggle('active', activeChips[key]);
    }

    function setStatus(msg) { document.getElementById('statusText').textContent = msg; }
    function showError(msg) { const el = document.getElementById('errorMsg'); el.textContent = msg; el.classList.add('show'); }
    function hideError() { document.getElementById('errorMsg').classList.remove('show'); }

    async function fetchRepoInfo(owner, repo) {
      const headers = { 'Accept': 'application/vnd.github.v3+json' };
      const [repoRes, contentsRes, languagesRes] = await Promise.all([
        fetch(`https://api.github.com/repos/${owner}/${repo}`, { headers }),
        fetch(`https://api.github.com/repos/${owner}/${repo}/contents`, { headers }),
        fetch(`https://api.github.com/repos/${owner}/${repo}/languages`, { headers }),
      ]);
      const repoData = await repoRes.json();
      const contents = contentsRes.ok ? await contentsRes.json() : [];
      const languages = languagesRes.ok ? await languagesRes.json() : {};
      return { repoData, contents, languages };
    }

    async function fetchSomeFiles(owner, repo, contents) {
      const codeExts = ['.js','.ts','.py','.java','.go','.rb','.php','.cs','.cpp','.c','.tsx','.jsx'];
      const files = (Array.isArray(contents) ? contents : [])
        .filter(f => f.type === 'file' && codeExts.some(e => f.name.endsWith(e)))
        .slice(0, 5);
      const results = [];
      for (const f of files) {
        try {
          const r = await fetch(f.download_url);
          if (r.ok) {
            const text = await r.text();
            results.push({ name: f.name, path: f.path, content: text.slice(0, 3000) });
          }
        } catch(e) {}
      }
      return results;
    }

    async function callClaude(prompt, apiKey) {
      const res = await fetch('https://api.anthropic.com/v1/messages', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'x-api-key': apiKey,
          'anthropic-version': '2023-06-01',
          'anthropic-dangerous-direct-browser-access': 'true'
        },
        body: JSON.stringify({
          model: 'claude-sonnet-4-20250514',
          max_tokens: 1500,
          messages: [{ role: 'user', content: prompt }]
        })
      });
      const data = await res.json();
      if (data.error) throw new Error(data.error.message);
      return data.content?.map(b => b.text || '').join('') || '';
    }

    function parseJSON(text) {
      try { return JSON.parse(text.replace(/```json|```/g, '').trim()); }
      catch(e) { return null; }
    }

    function renderBugs(bugs) {
      document.getElementById('bugCount').textContent = bugs.length + ' found';
      document.getElementById('bugsBody').innerHTML = bugs.length
        ? bugs.map(b => `<div class="issue-item"><div class="issue-file">${b.file||'unknown'}</div><div class="issue-text">${b.description}</div>${b.fix?`<div class="issue-fix">${b.fix}</div>`:''}</div>`).join('')
        : '<p class="prose">No bugs detected.</p>';
    }

    function renderQuality(issues) {
      document.getElementById('qualityScore').textContent = issues.length + ' issues';
      document.getElementById('qualityBody').innerHTML = issues.length
        ? issues.map(q => `<div class="issue-item"><div class="issue-file">${q.file||'general'}</div><div class="issue-text">${q.description}</div></div>`).join('')
        : '<p class="prose">Code quality looks good.</p>';
    }

    function renderSecurity(issues) {
      document.getElementById('securityScore').textContent = issues.length + ' risks';
      document.getElementById('securityBody').innerHTML = issues.length
        ? issues.map(s => `<div class="issue-item"><div class="issue-file">${s.file||'general'}</div><div class="issue-text">${s.description}</div></div>`).join('')
        : '<p class="prose">No obvious security issues found.</p>';
    }

    function renderReport(report, repoData) {
      document.getElementById('summaryGrid').innerHTML = `
        <div class="metric"><div class="metric-label">Stars</div><div class="metric-value">${repoData.stargazers_count ?? '—'}</div></div>
        <div class="metric"><div class="metric-label">Forks</div><div class="metric-value">${repoData.forks_count ?? '—'}</div></div>
        <div class="metric"><div class="metric-label">Language</div><div class="metric-value">${repoData.language || '—'}</div></div>
        <div class="metric"><div class="metric-label">Open issues</div><div class="metric-value">${repoData.open_issues_count ?? '—'}</div></div>`;
      document.getElementById('reportBody').innerHTML = `<p>${report}</p>`;
    }

    async function runReview() {
      hideError();
      const url = document.getElementById('repoUrl').value.trim();
      const apiKey = document.getElementById('apiKey').value.trim();

      if (!apiKey) { showError('Please enter your Anthropic API key.'); return; }
      if (!url) { showError('Please enter a GitHub repo URL.'); return; }

      const match = url.match(/github\.com\/([^\/]+)\/([^\/]+)/);
      if (!match) { showError('Could not parse the GitHub URL. Use format: https://github.com/owner/repo'); return; }
      const [, owner, repo] = match;
      const cleanRepo = repo.replace(/\/$/, '');

      document.getElementById('results').classList.remove('show');
      document.getElementById('statusBar').classList.add('show');
      document.getElementById('runBtn').disabled = true;

      try {
        setStatus('Fetching repository info...');
        const { repoData, contents, languages } = await fetchRepoInfo(owner, cleanRepo);
        if (repoData.message === 'Not Found') throw new Error('Repository not found or is private.');

        setStatus('Reading source files...');
        const files = await fetchSomeFiles(owner, cleanRepo, contents);

        const filesSummary = files.map(f => `### ${f.path}\n${f.content}`).join('\n\n---\n\n');
        const langList = Object.keys(languages).join(', ') || repoData.language || 'unknown';
        const context = `Repository: ${repoData.full_name}\nDescription: ${repoData.description || 'none'}\nLanguages: ${langList}\nStars: ${repoData.stargazers_count}\n\nFile samples:\n${filesSummary || 'No code files found in root.'}`;

        const sections = Object.keys(activeChips).filter(k => activeChips[k]);

        setStatus('Running AI analysis...');
        const prompt = `You are a senior code reviewer. Analyse this GitHub repository and return ONLY a valid JSON object (no markdown, no explanation) with these fields:
${activeChips.bugs ? '"bugs": array of up to 5 objects with {file, description, fix}' : ''}
${activeChips.quality ? '"quality": array of up to 5 objects with {file, description}' : ''}
${activeChips.security ? '"security": array of up to 4 objects with {file, description}' : ''}
${activeChips.report ? '"report": a 2-3 sentence daily summary of overall health, key risks, and recommended next action' : ''}

Context:
${context}

Return ONLY the raw JSON object, nothing else.`;

        const raw = await callClaude(prompt, apiKey);
        const data = parseJSON(raw);
        if (!data) throw new Error('AI returned an unexpected response. Please try again.');

        document.getElementById('statusBar').classList.remove('show');
        document.getElementById('results').classList.add('show');

        document.getElementById('card-bugs').style.display = activeChips.bugs ? '' : 'none';
        document.getElementById('card-quality').style.display = activeChips.quality ? '' : 'none';
        document.getElementById('card-security').style.display = activeChips.security ? '' : 'none';
        document.getElementById('card-report').style.display = activeChips.report ? '' : 'none';

        if (activeChips.bugs) renderBugs(data.bugs || []);
        if (activeChips.quality) renderQuality(data.quality || []);
        if (activeChips.security) renderSecurity(data.security || []);
        if (activeChips.report) renderReport(data.report || 'No summary generated.', repoData);

      } catch(e) {
        document.getElementById('statusBar').classList.remove('show');
        showError(e.message || 'Something went wrong. Please try again.');
      } finally {
        document.getElementById('runBtn').disabled = false;
      }
    }
  </script>
